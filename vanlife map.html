  <h2>My Sites</h2>

  <div id="site-list"></div>


  <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

  <script>

    const map = L.map('map').setView([31.2304, 121.4737], 9);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

      attribution: 'Map data © OpenStreetMap contributors'

    }).addTo(map);


    const siteList = document.getElementById('site-list');

    const form = document.getElementById('add-site-form');


    const sites = [];


    form.addEventListener('submit', function(e) {

      e.preventDefault();


      const name = document.getElementById('site-name').value;

      const lat = parseFloat(document.getElementById('lat').value);

      const lng = parseFloat(document.getElementById('lng').value);

      const category = document.getElementById('category').value;

      const description = document.getElementById('description').value;


      const newSite = { name, lat, lng, category, description };

      sites.push(newSite);


      // Add marker to map

      L.marker([lat, lng]).addTo(map)

        .bindPopup(`<b>${name}</b><br>${category}<br>${description}`).openPopup();


      // Add to list

      const card = document.createElement('div');

      card.className = 'site-card';

      card.innerHTML = `<strong>${name}</strong><br />

        <span class="tag">${category}</span> (${lat.toFixed(4)}, ${lng.toFixed(4)})<br />

        <small>${description}</small>`;

      siteList.appendChild(card);


      // Reset form

      form.reset();

    });

  </script>
